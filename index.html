            const target = document.getElementById(id);

            if(target) target.classList.remove('hidden');

            if(id === 'catalog-screen') renderCatalog();

            window.scrollTo(0, 0);

        }


        function startQuiz() {

            quizQueue = [];

            Object.keys(attributeQuestions).forEach(attrId => {

                attributeQuestions[attrId].forEach((q, idx) => {

                    quizQueue.push({ attrId, ...q, qIdxInAttr: idx });

                });

            });

            for (let i = quizQueue.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [quizQueue[i], quizQueue[j]] = [quizQueue[j], quizQueue[i]];

            }

            currentQIdx = 0;

            currentAnswers = [];

            showSection('quiz-screen');

            renderQuestion();

        }


        function renderQuestion() {

            const q = quizQueue[currentQIdx];

            if(!q) return;

            document.getElementById('choice-a').textContent = q.yes;

            document.getElementById('choice-b').textContent = q.no;

            document.getElementById('q-counter').textContent = `Question ${currentQIdx + 1} / 81`;

            document.getElementById('progress').style.width = `${((currentQIdx + 1) / 81) * 100}%`;

            const undoBtn = document.getElementById('undo-btn');

            undoBtn.style.opacity = currentAnswers.length > 0 ? "1" : "0";

            undoBtn.style.pointerEvents = currentAnswers.length > 0 ? "auto" : "none";

        }


        function answer(isYes) {

            const q = quizQueue[currentQIdx];

            currentAnswers.push({ ...q, isYes });

            currentQIdx++;

            if (currentQIdx >= 81) {

                calculateScoresAndRender();

                showSection('result-screen');

            } else {

                renderQuestion();

            }

        }


        function undo() {

            if (currentAnswers.length === 0) return;

            currentAnswers.pop();

            currentQIdx--;

            renderQuestion();

        }


        function calculateScoresAndRender() {

            const scores = {};

            traits.forEach(t => scores[t.id] = 0);

            currentAnswers.forEach(ans => {

                if (ans.isYes) scores[ans.attrId]++;

            });

            let maxId = 'A', minId = 'A', maxVal = -1, minVal = 10;

            Object.keys(scores).forEach(id => {

                if (scores[id] > maxVal) { maxVal = scores[id]; maxId = id; }

                if (scores[id] < minVal) { minVal = scores[id]; minId = id; }

            });

            if (maxId === minId) minId = Object.keys(scores).find(id => id !== maxId) || 'B';

            

            lastScores = scores;

            renderResult(maxId, minId, scores);

        }


        function recalculateAndScroll() {

            calculateScoresAndRender();

            window.scrollTo({ top: 0, behavior: 'smooth' });

        }

            }

        }
